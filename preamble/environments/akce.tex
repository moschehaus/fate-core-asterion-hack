\usepackage{enumitem}  % For customizing itemize/enumerate environments

% Custom environment for akce (icons as bullets)
\newenvironment{akce}
  {\begin{itemize}[
      label={},                % no default bullet
      align=parleft,           % align wrapped lines with text, not icon
      labelwidth=2cm,          % reserve space for the icon
      labelsep=0.5em,          % space between icon and text
      leftmargin=!,            % calculate total margin automatically
      itemsep=5.0em            % optional: spacing between items
    ]}
  {\end{itemize}}

% Icon + text item
\newcommand{\iconitem}[2]{%
    \item[\raisebox{-0.75\height}{\includegraphics[width=1.5cm]{#1}}] #2
}
